- $emacs ~/.emacs.d/init.el

#+begin_src
(global-set-key (kbd "C-c C-w") 'copy-lines)

(defun copy-lines(&optional arg)
  (interactive "p")
  (save-excursion
    (beginning-of-line)
    (set-mark (point))
    (next-line arg)
    (kill-ring-save (mark) (point))
  )
)
#+end_src

- 然后使用C-c C-w组合键来拷贝,也可在前面加C-u number参数来拷贝数行.
