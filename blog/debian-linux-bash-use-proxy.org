#+DESCRIPTION: Apt and Bash use Proxy

Bash 设置代理
#+begin_src
# add to ~/.bashrc
http_proxy=http://users:password@address:port
export http_proxy
#+end_src

Apt 设置代理
#+begin_src
# add to /etc/apt/apt.conf
Acquire::http::Proxy “http://address:port”;
#+end_src
